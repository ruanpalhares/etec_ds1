CREATE DATABASE BDEMPRESA;

USE BDEMPRESA;

CREATE TABLE DEPARTAMENTO(

IDDEP int not null primary key,
NOMEDEP varchar(70),
SIGLADEP varchar(5)

);

CREATE TABLE FUNCIONARIO(
IDFUNC INT NOT NULL PRIMARY KEY Auto_increment,
NOME VARCHAR(50),
CEP VARCHAR(9),
ENDERECO VARCHAR(70),
COMPLEMENTO VARCHAR(70),
NUMERO VARCHAR(10),
BAIRRO VARCHAR(50),
CIDADE VARCHAR(50),
UF varchar(2),
SALARIO DECIMAL(10,2),
IDDEP INT,
CONSTRAINT FOREIGN KEY(IDDEP) REFERENCES DEPARTAMENTO(IDDEP)
);
